// シェア機能（最小限の実装） function getShareText() { // 各ページの構造に合わせてセレクタを調整 const grade = document.querySelector('.grade-value')?.textContent || 'B'; const error = document.querySelector('.error-value')?.textContent || '±20'; const sessions = document.querySelector('.sessions-count')?.textContent || '8'; return `🎵 8va相対音感トレーニングで【${grade}級】獲得！\\n平均誤差: ${error}セント\\n${sessions}セッション完走！`; } // SNSシェア（URL生成のみ） function shareToTwitter(event) { event.preventDefault(); const text = encodeURIComponent(getShareText()); const url = encodeURIComponent(window.location.href); window.open(`https://twitter.com/intent/tweet?text=${text}&url=${url}`, '_blank'); } function shareToLine(event) { event.preventDefault(); const text = encodeURIComponent(getShareText() + '\\n' + window.location.href); window.open(`https://social-plugins.line.me/lineit/share?text=${text}`, '_blank'); } function shareToFacebook(event) { event.preventDefault(); const url = encodeURIComponent(window.location.href); window.open(`https://www.facebook.com/sharer/sharer.php?u=${url}`, '_blank'); } // コピー機能（CSSアニメーション付き） function copyShareText(event) { event.preventDefault(); const text = getShareText() + '\\n' + window.location.href; navigator.clipboard.writeText(text).then(() => { const btn = event.currentTarget; btn.classList.add('copied'); setTimeout(() => btn.classList.remove('copied'), 2000); }); }
